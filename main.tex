\section{QUESTION No-2.51(Inequalities)}
Solve:  x+y $\leq$ 6, x+y $\ge$ 4,
\section{Solution}
Let
-x-y $\ge$ -6,x+y $\ge$ 4.


Let $u_1 \ge 0, u_2 \ge 0$. 

This may be expressed as
\begin{align}
\vec{u} = \myvec{u_1\\u_2}\succeq \vec{0}
\end{align}
Now we have,
\begin{align}
  \myvec{-1&-1 \\ 1&1 }\vec{x} \succeq \myvec{-6 \\ 4}  + \vec{u} 
\end{align}
\begin{align}
\vec{x} = \myvec{-1 & -1 \\ 1 & 1}^{-1}\myvec{-6 \\ 4} +\myvec{-1 & -1 \\ 1 & 1}^{-1} \vec{u}
\\
\vec{x} = \frac{-1}{0}\myvec{-6 \\ 4}+\frac{-1}{0}\myvec{1&-1 \\ 1&1}\vec{u}
\\
\vec{x}=\myvec{\frac{6}{0} \\ \frac{-4}{0}} +\frac{-1}{0}\myvec{1&-1 \\ 1&1}\vec{u}
\end{align}
